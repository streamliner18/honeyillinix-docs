44  * Inputs: Number of an IRQ line

46  * Function: enables the specified IRQ line on the PICs */
47 void enable_irq(uint32_t irq_num) {
48  // BEGIN CP1.4
49  // REF P#13-14 Intel 8259A PIC datasheet, OCW1
50 
51  uint8_t int_reg_value; // holds mask
52 
53  if (irq_num < 8) { // if a IRQ number is for master
54  int_reg_value = inb(MASTER_8259_PORT+1); // read in current mask from master data port
55  int_reg_value = int_reg_value & ~(0x1 << irq_num); // set selected bit to zero
56  outb(int_reg_value, MASTER_8259_PORT+1); // send it back to master data port
57  }
58 
59  else if (irq_num >= 8 && irq_num < 16) { // if IRQ number is for slave
60  int_reg_value = inb(SLAVE_8259_PORT+1); // read in current mask from slave data port
61  int_reg_value = int_reg_value & ~(0x1 << (irq_num - 8)); // set selected bit to zero, rest to one
62  outb(int_reg_value, SLAVE_8259_PORT+1); // send it back to slave data port
63  }
